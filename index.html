import React, { useState, useEffect, useRef } from 'react';

import { initializeApp } from 'firebase/app';

import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';

import { getFirestore, doc, setDoc, onSnapshot, collection, query } from 'firebase/firestore';

import { 

  Calendar as CalendarIcon, 

  Plus, 

  Trash2, 

  Camera, 

  Image as ImageIcon, 

  Loader2, 

  X, 

  ChevronLeft, 

  ChevronRight,

  Archive,

  Star,

  CheckCircle2,

  Backpack

} from 'lucide-react';


// --- Firebase Configuration ---

const firebaseConfig = JSON.parse(__firebase_config);

const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);

const appId = typeof __app_id !== 'undefined' ? __app_id : 'daily-bag-diary';

const apiKey = ""; 


// --- UI Components ---

const Button = ({ children, onClick, variant = 'primary', className = '', disabled = false, isLoading = false }) => {

  const baseStyles = "px-4 py-3 rounded-2xl font-bold transition-all active:translate-y-1 shadow-[0_4px_0_0_rgba(0,0,0,0.1)] hover:brightness-110 disabled:opacity-50 disabled:cursor-not-allowed flex items-center justify-center gap-2";

  const variants = {

    primary: "bg-[#58cc02] text-white shadow-[#46a302]",

    secondary: "bg-[#1cb0f6] text-white shadow-[#1899d6]",

    ghost: "bg-white text-[#afafaf] border-2 border-[#e5e5e5] shadow-none",

    accent: "bg-[#ffc800] text-white shadow-[#e5a400]",

    outline: "bg-white text-[#58cc02] border-2 border-[#58cc02] shadow-none"

  };

  return (

    <button onClick={onClick} disabled={disabled || isLoading} className={`${baseStyles} ${variants[variant]} ${className}`}>

      {isLoading ? <Loader2 className="animate-spin" size={20} /> : children}

    </button>

  );

};


export default function App() {

  const [user, setUser] = useState(null);

  const [currentDate, setCurrentDate] = useState(new Date());

  const [view, setView] = useState('calendar');

  const [dailyLogs, setDailyLogs] = useState({});

  const [stickerLibrary, setStickerLibrary] = useState({});

  const [isAnalyzing, setIsAnalyzing] = useState(false);

  const [isCreatingSticker, setIsCreatingSticker] = useState(false);

  const [previewImage, setPreviewImage] = useState(null);

  const [detectedItems, setDetectedItems] = useState([]);


  const videoRef = useRef(null);

  const canvasRef = useRef(null);

  const fileInputRef = useRef(null);

  const dateKey = currentDate.toISOString().split('T')[0];

  const itemsToday = dailyLogs[dateKey] || [];


  // --- 1. Auth Initialization ---

  useEffect(() => {

    const initAuth = async () => {

      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

        await signInWithCustomToken(auth, __initial_auth_token);

      } else {

        await signInAnonymously(auth);

      }

    };

    initAuth();

    const unsubscribe = onAuthStateChanged(auth, setUser);

    return () => unsubscribe();

  }, []);


  // --- 2. Real-time Data Sync ---

  useEffect(() => {

    if (!user) return;


    // Sync Sticker Library

    const libraryDoc = doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'library');

    const unsubLibrary = onSnapshot(libraryDoc, (docSnap) => {

      if (docSnap.exists()) setStickerLibrary(docSnap.data().items || {});

    }, (err) => console.error("Library sync error:", err));


    // Sync Daily Logs

    const logsDoc = doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'logs');

    const unsubLogs = onSnapshot(logsDoc, (docSnap) => {

      if (docSnap.exists()) setDailyLogs(docSnap.data().entries || {});

    }, (err) => console.error("Logs sync error:", err));


    return () => {

      unsubLibrary();

      unsubLogs();

    };

  }, [user]);


  // --- 3. Persistent State Updates ---

  const saveLibrary = async (newLibrary) => {

    if (!user) return;

    await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'library'), { items: newLibrary });

  };


  const saveLogs = async (newLogs) => {

    if (!user) return;

    await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'logs'), { entries: newLogs });

  };


  // --- Helper & AI Logic ---

  const toggleItemForDay = async (id) => {

    const current = dailyLogs[dateKey] || [];

    const next = current.includes(id) ? current.filter(i => i !== id) : [...current, id];

    const newLogs = { ...dailyLogs, [dateKey]: next };

    setDailyLogs(newLogs);

    await saveLogs(newLogs);

  };


  const analyzeImage = async (base64Data) => {

    setIsAnalyzing(true);

    setView('analysis');

    try {

      const prompt = "Identify 3 items for a bag in this image. Return JSON: { items: [{ name: string }] }";

      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({

          contents: [{ parts: [{ text: prompt }, { inlineData: { mimeType: "image/png", data: base64Data.split(',')[1] } }] }],

          generationConfig: { responseMimeType: "application/json" }

        })

      });

      const result = await response.json();

      const parsed = JSON.parse(result.candidates[0].content.parts[0].text);

      setDetectedItems(parsed.items.map(i => ({ ...i, id: Math.random().toString(36).substr(2, 9) })));

    } catch (error) {

      setView('day-view');

    } finally {

      setIsAnalyzing(false);

    }

  };


  const createSticker = async (item) => {

    setIsCreatingSticker(true);

    try {

      const prompt = `Extract the '${item.name}' as a high-quality sticker with a thick white border.`;

      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`, {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({

          contents: [{ parts: [{ text: prompt }, { inlineData: { mimeType: "image/png", data: previewImage.split(',')[1] } }] }],

          generationConfig: { responseModalities: ['TEXT', 'IMAGE'] }

        })

      });

      const result = await response.json();

      const stickerBase64 = result.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;

      if (stickerBase64) {

        const id = Math.random().toString(36).substr(2, 9);

        const newSticker = { id, name: item.name, stickerUrl: `data:image/png;base64,${stickerBase64}` };

        const updatedLibrary = { ...stickerLibrary, [id]: newSticker };

        setStickerLibrary(updatedLibrary);

        await saveLibrary(updatedLibrary);

        await toggleItemForDay(id);

        setDetectedItems(prev => prev.filter(i => i.name !== item.name));

      }

    } finally {

      setIsCreatingSticker(false);

    }

  };


  const startCamera = async () => {

    setView('camera');

    try {

      const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } });

      if (videoRef.current) videoRef.current.srcObject = stream;

    } catch (err) {

      setView('day-view');

    }

  };


  const takePhoto = () => {

    if (videoRef.current && canvasRef.current) {

      const canvas = canvasRef.current;

      canvas.width = videoRef.current.videoWidth;

      canvas.height = videoRef.current.videoHeight;

      canvas.getContext('2d').drawImage(videoRef.current, 0, 0);

      const dataUrl = canvas.toDataURL('image/png');

      setPreviewImage(dataUrl);

      videoRef.current.srcObject.getTracks().forEach(t => t.stop());

      analyzeImage(dataUrl);

    }

  };


  // --- Rendering Helpers ---

  const renderCalendar = () => {

    const year = currentDate.getFullYear();

    const month = currentDate.getMonth();

    const firstDay = new Date(year, month, 1).getDay();

    const daysInMonth = new Date(year, month + 1, 0).getDate();

    const days = [];

    for (let i = 0; i < firstDay; i++) days.push(null);

    for (let i = 1; i <= daysInMonth; i++) days.push(i);


    return (

      <div className="space-y-6">

        <div className="flex items-center justify-between px-2">

          <h2 className="text-2xl font-black text-[#3c3c3c]">{year}년 {month + 1}월</h2>

          <div className="flex gap-2">

            <button onClick={() => setCurrentDate(new Date(year, month - 1, 1))} className="p-2 bg-white border-2 border-[#e5e5e5] rounded-xl"><ChevronLeft /></button>

            <button onClick={() => setCurrentDate(new Date(year, month + 1, 1))} className="p-2 bg-white border-2 border-[#e5e5e5] rounded-xl"><ChevronRight /></button>

          </div>

        </div>

        <div className="grid grid-cols-7 gap-2">

          {['일', '월', '화', '수', '목', '금', '토'].map(d => <div key={d} className="text-center text-xs font-bold text-[#afafaf]">{d}</div>)}

          {days.map((day, idx) => {

            const dKey = day ? `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}` : null;

            const hasItems = dKey && dailyLogs[dKey]?.length > 0;

            return (

              <div 

                key={idx} 

                onClick={() => day && (setCurrentDate(new Date(year, month, day)), setView('day-view'))}

                className={`aspect-square rounded-2xl flex flex-col items-center justify-center cursor-pointer border-2 transition-all ${day ? (hasItems ? 'bg-white border-[#58cc02] shadow-[0_4px_0_0_#58cc02]' : 'bg-white border-[#e5e5e5]') : 'border-transparent'}`}

              >

                {day && <span className="font-black text-sm">{day}</span>}

                {hasItems && <div className="w-1.5 h-1.5 bg-[#58cc02] rounded-full mt-1" />}

              </div>

            );

          })}

        </div>

      </div>

    );

  };


  if (!user) return (

    <div className="h-screen flex flex-col items-center justify-center gap-4 bg-[#f7f7f7]">

      <Loader2 className="animate-spin text-[#58cc02]" size={48} />

      <p className="font-bold text-[#afafaf]">데이터를 연결하는 중...</p>

    </div>

  );


  return (

    <div className="min-h-screen bg-[#f7f7f7] font-sans pb-24">

      <header className="bg-white border-b-2 border-[#e5e5e5] p-4 sticky top-0 z-50">

        <div className="max-w-md mx-auto flex justify-between items-center">

          <div className="flex items-center gap-2 cursor-pointer" onClick={() => setView('calendar')}>

            <div className="w-8 h-8 bg-[#58cc02] rounded-lg flex items-center justify-center text-white font-black">B</div>

            <h1 className="text-xl font-black text-[#4b4b4b]">BAG DIARY</h1>

          </div>

          {view !== 'calendar' && <button onClick={() => setView('calendar')} className="text-[#afafaf] font-bold text-sm">전체보기</button>}

        </div>

      </header>


      <main className="max-w-md mx-auto p-6">

        {view === 'calendar' && renderCalendar()}


        {view === 'day-view' && (

          <div className="space-y-6">

            <div className="flex items-center gap-4">

              <button onClick={() => setView('calendar')} className="p-2 border-2 rounded-xl"><ChevronLeft /></button>

              <h2 className="text-xl font-black">{currentDate.toLocaleDateString('ko-KR', { month: 'long', day: 'numeric' })}</h2>

            </div>


            <div className="grid grid-cols-2 gap-4">

              {itemsToday.map(id => (

                <div key={id} className="bg-white border-2 border-[#e5e5e5] rounded-3xl p-4 flex flex-col items-center relative shadow-[0_4px_0_0_#e5e5e5]">

                  <img src={stickerLibrary[id]?.stickerUrl} className="w-24 h-24 object-contain drop-shadow-md mb-2" />

                  <span className="font-bold text-sm">{stickerLibrary[id]?.name}</span>

                  <button onClick={() => toggleItemForDay(id)} className="absolute -top-2 -right-2 bg-[#ff4b4b] text-white p-1 rounded-full"><X size={12} /></button>

                </div>

              ))}

            </div>


            {itemsToday.length === 0 && (

              <div className="border-4 border-dashed border-[#e5e5e5] rounded-[2rem] py-16 flex flex-col items-center text-[#afafaf] gap-4">

                <Backpack size={48} className="opacity-20" />

                <p className="font-bold">오늘 가방에 무엇을 담았나요?</p>

                <Button onClick={startCamera} variant="outline">기록하기</Button>

              </div>

            )}


            {Object.keys(stickerLibrary).length > 0 && (

              <div className="pt-6 border-t-2">

                <h3 className="font-black mb-3">보관함에서 가져오기</h3>

                <div className="flex gap-3 overflow-x-auto pb-4 px-1 no-scrollbar">

                  {Object.values(stickerLibrary).map(sticker => (

                    <div 

                      key={sticker.id}

                      onClick={() => toggleItemForDay(sticker.id)}

                      className={`flex-shrink-0 w-20 h-24 bg-white border-2 rounded-2xl flex flex-col items-center justify-center p-2 transition-all ${itemsToday.includes(sticker.id) ? 'border-[#58cc02] bg-[#f2ffeb]' : 'border-[#e5e5e5]'}`}

                    >

                      <img src={sticker.stickerUrl} className="w-12 h-12 object-contain" />

                      <span className="text-[10px] font-bold mt-1 text-center truncate w-full">{sticker.name}</span>

                    </div>

                  ))}

                </div>

              </div>

            )}

          </div>

        )}


        {view === 'camera' && (

          <div className="fixed inset-0 z-[100] bg-black flex flex-col items-center justify-center p-4">

             <div className="relative w-full max-w-md aspect-[3/4] bg-[#333] rounded-[2.5rem] overflow-hidden border-4 border-[#58cc02]">

                <video ref={videoRef} autoPlay playsInline className="w-full h-full object-cover" />

                <div className="absolute inset-x-0 bottom-8 flex justify-center items-center gap-8 px-6">

                  <button onClick={() => setView('day-view')} className="p-4 bg-white/20 rounded-full text-white"><X size={32} /></button>

                  <button onClick={takePhoto} className="w-20 h-20 bg-white rounded-full border-8 border-[#58cc02]" />

                </div>

             </div>

             <canvas ref={canvasRef} className="hidden" />

          </div>

        )}


        {view === 'analysis' && (

          <div className="space-y-6 animate-in zoom-in-95">

             <h3 className="font-black text-xl text-[#3c3c3c]">발견된 아이템</h3>

             <div className="aspect-video bg-[#f0f0f0] rounded-3xl overflow-hidden relative border-2 border-[#e5e5e5]">

               {previewImage && <img src={previewImage} className={`w-full h-full object-cover ${isAnalyzing ? 'animate-pulse grayscale' : 'opacity-40'}`} />}

               {isAnalyzing && (

                 <div className="absolute inset-0 flex flex-col items-center justify-center text-[#58cc02] gap-2">

                   <Loader2 className="animate-spin" size={40} />

                   <p className="font-bold">AI 분석 중...</p>

                 </div>

               )}

             </div>

             <div className="grid grid-cols-2 gap-3">

               {detectedItems.map(item => (

                 <Button key={item.id} variant="accent" onClick={() => createSticker(item)} isLoading={isCreatingSticker}>

                   {item.name} 스티커화

                 </Button>

               ))}

             </div>

          </div>

        )}

      </main>


      {view !== 'camera' && (

        <nav className="fixed bottom-0 left-0 right-0 bg-white border-t-2 border-[#e5e5e5] p-4 z-50">

          <div className="max-w-md mx-auto flex gap-4">

            <Button className="flex-1" onClick={view === 'calendar' ? () => setView('day-view') : startCamera}>

              {view === 'calendar' ? '오늘의 가방 쓰기' : '새로운 아이템 촬영'}

            </Button>

            {view === 'day-view' && (

              <Button variant="outline" onClick={() => fileInputRef.current.click()}><ImageIcon /></Button>

            )}

            <input type="file" className="hidden" ref={fileInputRef} accept="image/*" onChange={(e) => {

               const file = e.target.files[0];

               if(file) {

                 const r = new FileReader();

                 r.onload = (ev) => { setPreviewImage(ev.target.result); analyzeImage(ev.target.result); };

                 r.readAsDataURL(file);

               }

            }} />

          </div>

        </nav>

      )}

    </div>

  );

}
